export async function handleCredentialData(uuid: string) {
  // Awaiting response from /sharedCredentials API call.
  const response: SharedCredentials = (await sharedCredentials(uuid)) as SharedCredentials;

  // Extracting the `credentials` from the `response`.
  const credentials: Credential[] = response.credentials;

  // Creating a new Map to hold individual credentials, keyed on their types.
  const map = new Map<credentialsTypes, ICredentialData>();

  // Iterating over each `credential` in the `credentials` array.
  for (const credential of credentials) {
    // Checking if the `credential.type` is a valid credential type.
    if (isCredentialsType(credential.type)) {
      // If valid, we add this credential to our `map` with its type as the key and data as the value.
      map.set(credential.type, credential.data);
    } else {
      logger.error(`Invalid credential type: ${credential.type}`);
    }
  }

  /**
   * A generic function to extract the value of a specific property from a credential from json response from the /sharedCredentials API call.
   *
   * @param key - The key used to fetch the credential item from the map.
   * @param property - The property of which value is needed.
   *
   * @returns The value of the specified `property` from the identified credential item.
   * If no such item/property exists, or if the resulting value is not valid (as determined by the `isValidValue()` function), `undefined` is returned instead.
   */
  function getValue<T = string>(key: credentialsTypes, property: keyof ICredentialData): T | undefined {
    // Get the credential item corresponding to the provided 'key'
    const item = map.get(key);

    // Access the provided 'property' of the retrieved 'item', if the item exists
    const propValue = item && item[property];

    // Check if 'propValue' is valid using 'isValidValue()' function.
    // If it is, return 'propValue' casted to T. If not, return 'undefined'.
    return isValidValue(propValue) ? (propValue as T) : undefined;
  }

  const firstName = getValue('FirstNameCredential', 'firstName');
  const lastName = getValue('LastNameCredential', 'lastName');
  const email = getValue('EmailCredential', 'email');
  const address = getValue('AddressCredential', 'address');
  const phone = getValue('PhoneCredential', 'phone');

  logger.debug({ firstName, lastName, email, address, phone });

  // validate the required credential values are present, as expected
  if (!firstName || !lastName || !email) {
    throw new Error('Missing required credentials');
  }

  return { firstName, lastName, email, address, phone };
}
